from jinja2 import Template as JinjaTemplate

def _render_final_html(self, raw_html: str, snapshot_data: dict):
    # 1. Sử dụng Jinja2 để render toàn bộ các mảng và key-value
    # snapshot_data = {
    #    "GuestName": "Nguyen Van A",
    #    "RoomServices": [{"Date": "26/03", "Description": "Room charge", "Amount": 2000000}],
    #    "OtherServices": [{"ServiceName": "Coca Cola", "Qty": 2, "Price": 60000}]
    # }
    
    try:
        tpl = JinjaTemplate(raw_html)
        html_rendered = tpl.render(**snapshot_data)
    except Exception as e:
        html_rendered = raw_html # Trả về bản thô nếu lỗi render
        print(f"Jinja2 Error: {e}")

    # 2. Sau khi Jinja2 chạy xong, ta mới cấy vùng ký (Mapping thủ công)
    # Vì vùng ký chứa code Javascript, để Jinja2 render có thể bị lỗi cú pháp
    html_final = html_rendered.replace("{{GuestSignatureImg}}", 
        '<div class="sig-placeholder" onclick="SIGN.actions.openPad(\'guest\')">'
        '<img id="img-guest-sig" src="" style="display:none; max-height:80px;" />'
        '<span class="placeholder-text">Chạm để ký / Touch to sign</span></div>')

    html_final = html_final.replace("{{ReceptionSignatureImg}}", 
        '<div class="sig-placeholder" onclick="SIGN.actions.openPad(\'reception\')">'
        '<img id="img-recep-sig" src="" style="display:none; max-height:80px;" />'
        '<span class="placeholder-text">Chạm để ký / Staff Sign</span></div>')
        
    return html_final
